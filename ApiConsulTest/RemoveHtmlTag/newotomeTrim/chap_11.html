ในช่วงเทศกาลปีใหม่ ในเมืองหลวงจะมีการฉลองกันถึง 7 วัน บรรดาดยุคผู้ครองแคว้นทั้ง 12 บรรดาขุนนาง เชื้อพระวงศ์ทั่วอาณาจักร ต่างต้องพาครอบครัวมาร่วมงานถวายพระพรพระราชาราชินี ทั่วเมืองหลวงประดับธงเฉลิมฉลอง ผู้คนแต่งตัวสวยงาม  จากแคว้นเทรลิสเดินทางถึงเมืองหลวงใช้เวลาประมาณ 2 วันหากเดินทางด้วยรถม้าชั้นดีและเร่งด้วยพลังเวทลมซึ่งอาจใช้ด้วยตัวนักเวทเองหรือเวทที่เก็บในลูกแล้วที่วางขาย แต่หากเดินทางแบบคนทั่วไปต้องใช้เวลาถึง 4 วันทีเดียว เลดี้มาร์ธาเตรียมเสื้อผ้าพร้อม ในการเดินทาง เรนิสนั่งในรถม้ากับท่านพ่อท่านแม่ ส่วนเธมม์ นั่งมากับท่านอาจารย์อัศวินและเลดี้มาร์ธาเนื่องจากตลอดทางยังต้องมีการทบทวนเรื่องมารยาทที่จริงปกติต้องแยกกันนั่งตามระบบความปลอดภัย “ท่านพ่อ ลูกอยากนั่งด้วย” เท่านั้นเรียบร้อย ท่านพ่อใจอ่อนกับเธอเสมอ อากาศภายนอกรถม้าหนาวขึ้นเรื่อยๆ บางบริเวณเมื่อออกจากแคว้นเทรลิสเริ่มมีหิมะตก แต่ในรถม้ายังอุ่นสบายด้วยเวทที่คล้ายกับการใช้ฮีทเตอร์ เรนิสนั่งคิดไปเรื่อยๆเรื่องของขวัญปีใหม่ที่เตรียมให้คนในครอบครัว ของท่านพ่อเธอเตรียมลูกแก้วเวทที่เปิดออกมาเป็นดอกทิวลิปร้องเพลงอวยพรให้ เธอคิดค้นรูปแบบเวทขึ้นมาเอง กะว่าถ้าท่านพ่อชอบ ในอนาคตเธออาจจะทำขาย สำหรับท่านแม่ เธอเตรียมสร้อยมุกออกประกายเงิน เข้ากับสีผมของท่านแม่เลยทีเดียว ตรงกลางห้อยจี้เพชรล้อมด้วยมุกอีกที ล้อมด้วยไข่มุกสีเงินเช่นกัน สร้อยเส้นนี้ราคาแพงไม่น้อย หลังจากเธอกับเลดี้มาร์ธาและสาวใช้ช่วยกันเลือก ทุกคนลงความเห็นว่าเส้นนี้สวยที่สุด เธอลงทุนอ้อนท่านพ่อครึ่งวันเพื่อขอเงินมาซื้ออยู่หลายเหรียญทอง สำหรับเลดี้มาร์ธาแสนดี เธอเตรียมเข็มกลัดพลอยอันหนึ่งที่ท่านแม่ช่วยเลือกให้ แน่ล่ะ เธอไม่ถนัดเรื่องเสื้อผ้าเครื่องประดับ ทุกวันนี้เสื้อผ้าเป็นเลดี้มาร์ธาจัดหามาให้ ส่วนการเลือกเสื้อผ้าในทุกเช้าเป็นงานของแอนนาว่าจะใส่อะไรกับอะไร เธอมีแค่หน้าที่ใส่แค่นั้นเอง นอกจากนี้ยังมีของขวัญต่างๆสำหรับอาจารย์เธอเลือกเป็นหนังสือ แอนนาได้เสื้อกันหนาวที่เลดี้มาร์ธาช่วยเลือก  และสำหรับพี่ชายที่บังเอิญได้มาใหม่ เธอสั่งตัดเสื้อคลุมกันหนาวที่กันหิมะได้จากผ้าสีดำที่เลดี้มาร์ธาเลือก ที่ไม่ซื้อขนมให้เพราะอยากให้เขาลดน้ำหนัก นี่ถ้าขาดเลดี้มาร์ธา เธอคงแย่แน่นอน“เรนิส ลูกเตรียมอะไรให้เจ้าชายซีริล” ท่านแม่เอ่ยขึ้นมาขณะที่เรนิสกำลังใจลอย ตายจริง! เธอลืมไปว่ามีคู่หมั้นที่ต้องให้ของขวัญ ถ้าจะแวะซื้อของข้างทางให้เจ้าชายเห็นจะไม่ดีแน่ “ถ้าลืมก็ไม่เป็นไรนะ เดี๋ยวเข้าเมืองหลวงก็รีบหาซักชิ้น” ท่านพ่อพูดเหมือนไม่ใช่เรื่องสำคัญ “ท่านพ่อรู้ได้อย่างไรคะว่าลูกลืม” “หึๆ ก็สีหน้าลูกแสดงออกมาชัดขนาดนั้น” ท่านพ่อหัวเราะเบาๆ เรนิสหันไปมองท่านแม่ที่อมยิ้มขำเธอเช่นกัน “พึ่งหมั้นกันแค่เดือนกว่าๆ เรนิสลืมคู่หมั้นซะแล้ว” เรนิสยิ้มแห้งๆกลับไป ความจริงแล้วอยากจะลืมไปตลอดเลย ไม่ก็หวังว่าเขาจะหายไปจากชีวิตของเธอแบบพระเอกหมายเลข 4  ในบรรดาผู้ชายทั้งหมดในเกม นับแล้วมีเขากับอเล็กซิสนี่แหละที่เป็นอันตรายกับเธอมากกว่าคนอื่นๆ คฤหาสน์เทรลิสในเมืองหลวงเวลานี้ปกคลุมด้วยหิมะบางๆ เวลาเดินต้องระวังลื่นอย่างมาก เมื่อเข้าในตัวบ้าน อากาศข้างในจะอบอุ่นด้วยเวทย์ เรนิสทักทายเธมม์ที่มองรอบๆอย่างตื่นเต้น “เป็นยังไงบ้างเธมม์ การเดินทางสบายไหม” ตอนนี้เรนิสไม่เรียกพี่ชายแล้วแต่เรียกตรงๆ เพื่อเตรียมฝึกให้นายหมูชินกับการเป็นเบ๊ “ก็ดีเรนิส ตอนนี้ท่องจำมารยาทบนโต๊ะได้หมดละ เสียดายระหว่างทางไม่ได้เปิดหน้าต่างออกดูเพราะมันหนาว” “อือ จริงสิ นี่เป็นครั้งแรกที่นายมาเมืองหลวงรึเปล่า” “ใช่ ฉันยังอยากดูรอบๆเลย ตั้งแต่เกิดมาพึ่งเคยมาเมืองหลวงนี่แหละ” “ช่วงนี้อากาศหนาวแล้วพอดีเป็นเทศกาล เค้าปิดร้านค้ากันฉลอง นอกจากไปดูบรรยากาศแล้ว ไม่มีอะไรขายหรอกนะ เอาไว้มาตอนกลางปีหรือช่วงที่อุ่นกว่านี้จะพาเดินเที่ยว” “ขอบคุณมากเรนิส เธอดีกับฉันจริงๆเลย” เธมม์ยิ้มกว้างให้เรนิส พูดถึงร้านค้าปิด เรนิสยังไม่ได้เตรียมอะไรให้คู่หมั้นเลย พอนึกได้เรนิสจึงรีบเก็บตัวเข้าห้อง สุดท้ายแล้วเธอตัดสินใจลูกแก้วดอกทิวลิปร้องเพลงอวยพรแบบเดียวกับที่ให้ท่านพ่อนี่แหละ ใช้เวลาสัก 1 ชั่วโมงเธอก็สร้างเสร็จ ดอกทิวลิปเลือกเป็นสีน้ำเงินซึ่งคิดว่าเป็นสีโปรดของเจ้าชาย นอกจากนี้เธอยังไปรื้อค้นของที่มีเก็บไว้ นอกจากเจ้าชายซีริลแล้ว คงต้องให้เจ้าชายซิลบาสเตียน ท่านชายแวน และท่านหญิงวีราเนสด้วย เพราะถือเป็นเพื่อนกันแล้ว เธอหยิบแผนที่อาณาจักรฟานส์ที่ท่านพ่อให้มาเตรียมให้กับเจ้าชายซิลบาสเตียน ปากกาเวทย์แท่งใหม่ที่ยังไม่ได้ใช้ให้แวน แล้วก็ริบบิ้นผูกผมอันใหม่สีขาวเป็นลูกไม้ปักด้วยมุกเล็กๆที่มาร์ธาเลือกมาให้เธอใช้แต่ยังไม่ได้ใช้ให้วีราเนส ว่าแล้วก็นำของทั้งหมดไปฝากแอนนาห่อของขวัญ ก็บรรดางานเย็บผ้า การห่อของต่างๆเธอไร้ความสามารถสิ้นเชิงนี่แหละ เพื่อนเยอะก็สิ้นเปลืองมากยังดีไม่ได้ซื้อเอง ไม่งั้นเสียดายเงินแย่เลย เธอถอนหายใจด้วยความเสียดาย หวังว่าของขวัญที่คนอื่นให้เธอมาคงจะเป็นของมีค่าเหมือนกัน…หลังจากเข้าถวายพระพรพระราชาและราชินีแล้ว ท่านพ่อแนะนำเธมม์ เทรลิสว่าเป็นบุตรบุญธรรมคนใหม่ ท่ามกลางความแปลกใจและสงสัยของบรรดาขุนนางและเชื้อพระวงศ์ต่างๆอย่างมาก ก็แน่ล่ะสิ ระดับท่านดยุค รับบุตรบุญธรรมทั้งที คนย่อมคาดหวังเธมม์ได้รับการแต่งตั้งเป็นลอร์ดเธมม์ เทรลิส อย่างเป็นทางการ ตำแหน่งลอร์ดหรือเลดี้ ไม่มีอะไรมาก สามารถได้มาแต่กำเนิดโดยเกิดในครอบครัวชนชั้นสูงหรือขุนนางระดับสูงพอจะได้แต่งตั้งโดยอัตโนมัติ ถัดมาคือการรับเป็นบุตรบุญธรรมในครอบครัวชนชั้นสูง และสุดท้ายคือการได้รับการแต่งตั้งจากการมีความดีความชอบ หลังได้รับการแต่งตั้งแล้วจะถือว่ากลายเป็นชนชั้นสูงทันที อย่างเช่นคุณนางเอก ก่อนได้แต่งงานกับบรรดาเจ้าชายคือจะมีความดีความชอบได้รับแต่งตั้งเป็นเลดี้ก่อนสำหรับทายาทโดยตรงของผู้ครองแคว้นที่เป็นดยุกหรือดัชเชส จะเรียกว่าแกรนด์ลอร์ด และแกรนด์เลดี้ หรือท่านหญิงและท่านชายก็ได้ แต่บางครั้งคนทั่วไปหรือขุนนางใหม่ๆอาจจะสับสนเรียกเป็นลอร์ดและเลดี้ซึ่งถือว่าเป็นการลดตำแหน่งลง แต่ส่วนใหญ่แล้วบรรดาแกรนด์ลอร์ดและแกรนด์เลดี้มักจะไม่ถือสา หากถือสาก็มักเก็บไว้ในใจเพื่อแสดงความใจกว้างให้สังคมเห็น นายหมูอ้วนหรือตอนนี้เป็นลอร์ดเธมม์ มีใบหน้าแดงก่ำด้วยความเก้อเขินและตื่นเต้นเด็กๆอย่างพวกเธอถูกแยกออกมาเพื่อให้ผู้ใหญ่คุยกัน เธอมอบของขวัญให้บรรดาเจ้าชาย และท่านหญิง ท่านชาย“พี่ชายคนใหม่ของเธอน่าสนใจจริงๆ” วีราเนสกระซิบข้างหูเธอ สีหน้าเยาะเย้ย “ทั้งแคว้นเธอหาเด็กๆดีๆไม่ได้เลยหรือไง หรือมีแต่นายนี่”“เธอเข้าใจอะไรผิดหรือเปล่า” เรนิสแย้งกลับไป “ลองเอาพี่ชายเธอมาแข่งกับเธมม์สิ รับรองว่าสู้ไม่ได้”วีราเนสสีหน้าประหลาดใจ “เธอหมายความว่า ลอร์ดเธมม์มีความสามารถด้านการต่อสู้อย่างนั้นหรือ”“อือ ใช่ ลองเอาพี่ชายเธอมาแข่งมวยปล่ำกับเขาสิ…รับรอง พี่ชายเธอแบนเหมือนผืนพรมแน่นอน” เรนิสพูดอย่างมั่นใจ และถ้ามาแข่งกิน แวนก็ไม่มีทางชนะอีกนั่นแหละวีราเนสยังคงงุนงงสับสน เรนิสใช้โอกาสนี้ขอตัว เอาตัวเองและเธมม์หนีกลับ โดยอ้างว่าเหนื่อยจากการเดินทางต้องการพักผ่อนก่อนงานเต้นรำคืนนี้ แต่ความจริงแล้วเธอแอบได้ยินเสียงท้องของลอร์คคนใหม่ร้องจ๊อกๆ เธมม์ถูกเลดี้มาร์ธาจำกัดอาหารเพื่อรูปลักษณ์จะได้ดูดีขึ้น เรนิสเริ่มกลุ้มใจว่าเธมม์จะหน้ามืดเป็นลมด้วยความหิวไปซะก่อนจะหล่อหรูมีสไตล์อย่างที่เลดี้มาร์ธาหวัง ขณะอยู่บนรถม้า เธอยื่นขนมปังกรอบชิ้นเล็กให้เธมม์กินรองท้องก่อนที่จะเป็นลมด้วยความหิว “เรนิส เธอช่างแสนดีเหลือเกิน” เธมม์ทำราวกับขนมปังกรอบชิ้นนั้นเป็นของมีค่าหาไม่ได้อีกแล้วเรนิสถอนหายใจด้วยความท้อ เธอต้องเอาเธมม์กลับไปซ้อมใหญ่การเต้นรำก่อนออกงานจริงคืนนี้ หวังว่าขนมปังกรอบชิ้นนี้จะทำให้เธมม์มีแรงซ้อมเต้นรำอีกเหตุผลที่ต้องรีบกลับคือกลัวเธมม์ถูกจับผิดแล้วหลุดอะไรที่น่าขายหน้าออกมาอย่างน้อยๆก็เรื่องท้องร้องนี่แหละ ถ้าอยู่ต่อต้องมีคนได้ยินแน่ๆ ช่วยไม่ได้ เธอเป็นคนทำให้ท่านพ่อรับเธมม์เข้ามา อย่างน้อยเธอต้องรับผิดชอบเธมม์และตระกูลเทรลิสไม่ให้เกิดเรื่องขายหน้าขึ้นบ่ายนั้นทั้งบ่ายเธอฝึกเต้นรำกับเธมม์ เธมม์ทำได้พอใช้ทีเดียวถ้าไม่เปลี่ยนคู่เป็นคนอื่นและไม่มีท่าพลิกแพลง ดังนั้นเธอและเธมม์จึงตกลงกันว่าเธมม์จะเต้นรำกับเรนิสแล้วไม่ไปขอใครเต้นอีก “นายห้ามสบตาผู้หญิงคนไหนในงาน” เธอยื่นคำขาด “ถึงแม้ว่าผู้หญิงคนนั้นจะถือจานขนมในมือก็ตาม”โดยมารยาทแล้วผู้หญิงจะไม่มาขอผู้ชายเต้นรำอยู่แล้ว ยกเว้นว่าสนใจผู้ชายคนนั้นจะเดินมาเปรยว่าอยากเต้นรำ ถ้าฝ่ายชายเป็นสุภาพบุรุษก็จะรักษาหน้าให้ฝ่ายหญิงโดยเชิญฝ่ายหญิงเต้นรำ แต่หากฝ่ายชายเกรียนเกินไปก็จะปฏิเสธ ซึ่งจะทำให้ฝ่ายหญิงอับอายอย่างยิ่งและถือเป็นการเสียมารยาทอย่างมาก เมื่อประเมินจากรูปลักษณ์ของเธมม์แล้วคงยากที่จะมีเด็กหญิงคนไหนมาอยากเต้นรำด้วยล่ะนะ อีกอย่างพวกเธอยังเป็นเด็กๆกันอยู่เลย นอกจากจะมีคนอยากกลั่นแกล้งลองของฉากห้องโถงโอ่อ่าที่ใช้สำหรับงานเต้นรำ ประดับประดาไปด้วยลูกแก้วเวทสีต่างๆ ทำให้ห้องดูสวยงามยิ่งใหญ่มาก เรนิสถอนหายใจอีกครั้ง เธอไม่ชอบห้องโถงนี้เลย นี่เป็นฉากจบของเกมในหลายๆรูท อย่างน้อยก็เป็นฉากที่ซีริลขอถอนหมั้นล่ะน่า เธอคิดถึงสมุดเล่มนั้นที่เธอล็อกอย่างแน่นหนา พระราชาผู้สง่างามเปิดฟลอร์เต้นรำกับพระราชินี ทั้งคู่เต้นรำได้งดงามอย่างยิ่ง ต่อมาบรรดาขุนนางบรรดาศักดิ์ ท่านดยุคและดัชเชสต่างทยอยออกมาเต้นรำ เจ้าชายซีริลโค้งให้เรนิสแล้วพาเธอออกไปเต้นรำ ตามด้วยเจ้าชายซิลบาสเตียนและท่านหญิงวีราเนส ส่วนเธมม์ยืนกับเลดี้มาร์ธาเพื่อรอเธอมาเต้นรำด้วยในเพลงถัดไป“ขอบคุณสำหรับของขวัญปีใหม่มาก เป็นของขวัญที่แปลกตาและน่ารักมากเลย” เจ้าชายซีริลยิ้มให้เรนิสด้วยสีหน้าอ่อนโยน“เจ้าชายทรงโปรดไหมเพคะ” เรนิสถามด้วยรอยยิ้มที่พยายามปั้นออกมา“อืม ชอบสิ เรนิสให้มาต้องชอบอยู่แล้ว” เสียงเจ้าชายช่างอ่อนโยนชวนฝันน่าประทับ แต่พอนึกถึงรูทหายนะ ไม่ว่าจะฉากลอบสังหาร ยาดื่มไวน์ผสมยาพิษแล้ว รอยยิ้มของเรนิสที่ค้างอยู่แต่แรกกลับมีการกระตุกที่มุมปากเวลาคิดถึงว่าเด็กชายรูปหล่อตรงหน้าจะวางแผนฆ่าเธอยังไงได้บ้างก็ฟินกับความหล่อระดับเทพนี่ไม่ลงซะละ“แล้วเรนิสชอบของขวัญที่ฉันให้ไหม” จริงสิเจ้าชายให้ของขวัญมาแต่เธอมัวแต่ซ้อมเต้นรำกับเธมม์ ลืมเปิดดู“หม่อมฉันจะเปิดดูคืนนี้เพคะ แต่เจ้าชายประทานอะไรมา หม่อมฉันก็ชอบทั้งนั้นแหละ” เรนิสพยายามคงรอยยิ้มหวานและคำพูดออดอ้อนตราบใดที่ไม่วางแผนฆ่าเธอ เธอก็พอใจละ ของขวัญอะไรนั่นไม่ต้องให้ก็ได้“ได้ยินท่านดยุคบอกว่าดอกทิวลิปร้องเพลง เรนิสทำขึ้นมาเอง”“ก็เป็นเวทพื้นฐานของธาตุไม้ แล้วใส่เวทร้องเพลงเข้าไปผสมเพคะ การร่ายเวทย์สร้างลูกแก้วทำไม่ได้ยากเท่าไหร่ แต่ยากที่การออกแบบเวทให้ดูกลมกลืนกัน”“นับว่าเธอมีพรสวรรค์นะเรนิส”“ขอบพระทัยเพคะ” โชคดีที่เพลงสิ้นสุด อย่างน้อยเธอไม่ต้องมาทนปั้นรอยยิ้มปลอมๆ การทำหมางเมินกับเจ้าชายจะทำให้คนสงสัยได้อีก รูทหายนะ ยังไงก็คือรูทหายนะจะหล่อ จะดีแค่ไหน มันก็รูทหายนะอีกนั่นแหละ เธอย้ำกับตัวเอง อย่าได้ถูกเขาหลอกลวงด้วยภาพความสุภาพอ่อนโยนหลังเพลงสิ้นสุด เรนิสย่อตัวทำความเคารพก่อนที่จะแยกตัวเดินไปหาเธมม์ ตามที่ตกลงกันไว้เธอจะมาเต้นรำกับเธมม์อีกเพลงให้คนรู้ว่าเธมม์เต้นรำได้ เธอเห็นวีราเนสกระซิบกับเพื่อนสาวคนหนึ่ง รู้สึกจะเป็นลูกขุนนางสักคนที่เป็นสมุนของวีราเนส ตามพล็อตเลย วีราเนสไม่ออกหน้าเองเท่าไหร่แต่มักใช้เบ๊ทำงานแทน ขอเดาว่าหล่อนคนนั้นคงจะเดินมาบอกเธมม์ว่าอยากเต้นรำ เมื่อเห็นว่าเบ๊ของคุณวิลเลี่ยนเนสกำลังเดิมมาหาเธมม์ เธอจึงเร่งเดินไปหาก่อน เธมม์โค้งคำนับเธอตามธรรมเนียมและเธอย่อตัวตอบ ทั้งคู่เริ่มเต้นรำตามแบบที่ซักซ้อมไว้ เธอมองรอบๆเห็นสายตาหลายคู่จับจ้องมา หลายคนพากันจับผิดเธมม์ว่าจะทำอะไรขายหน้ารึเปล่า เธอเห็นวีราเนสเต้นรำกับเจ้าชายซีริลคู่หมั้นของเธอขณะที่ซิลบาสเตียนหายไป เหลือเชื่อ ซิลบาสเตียนปลีกตัวออกจากปลิงอย่างวีราเนสได้อย่างไร เธมม์เต้นตามที่ฝึกซ้อมได้ดี งานนี้นับว่าไม่ขายหน้า พอเต้นเสร็จพวกเธอควรรีบเผ่นกลับก่อนที่ใครจะจับผิดได้ ทันทีที่เพลงจบ เพื่อนหญิงตัวร้ายคนนั้นทำท่าจะเดินมาหาเธมม์อีก“เธมม์ มีคนจะเดินมาหาเธอ เราเต้นรำกันต่อแล้วหลบออกทางประตูนั้น” เรนิสตัดสินใจบอกเธมม์ที่จริงปกติงานเต้นรำถ้าไม่ใช้คู่สามีภรรยา คู่หมั้นหรือคู่รัก มักไม่เต้นรำกันเพลงที่2 ต่อ ยกเว้นถูกใจกันอยากสนิทสนม หรืออยากพูดคุยกันต่อ ซึ่งจะทำให้เป็นที่จับตามอง ในกรณีที่เป็นพี่น้องก็มักจะเกิดจากการต้องการกันท่าไม่ให้เพศตรงข้ามเข้าหาพี่น้องของตนเอง ในกรณีเรนิสกับเธมม์ เป็นพี่น้องต่างสายเลือดกันมาเกือบเดือนก็นับเป็นพี่น้องได้ เธอคิดเข้าข้างตัวเองพอเพลงจบ เธอจูงมือเธมม์หลบออกจากงาน ขึ้นรถม้าหนีกลับไปอย่างแนบเนียน  หลังจากอยู่ในเมืองหลวงครบ7วัน ครอบครัวเทรลิสรีบกลับแคว้นโดยไม่รั้งรอ ในระหว่างที่อยู่เมืองหลวง เรนิส ปฏิเสธงานน้ำชาของสมาคมเด็กโดยไม่ใยดีด้วยเหตุผลว่าเป็นหวัดหลังการเดินทาง และเธมม์ก็ติดหวัดการเมืองจากเธอด้วยเธอใช้ข้ออ้างนี้ปฏิเสธบัตรเชิญไปทั้งหมด “เรนิส เธอช่างแสนดีจริงๆ” เธมม์มองเธออย่างซาบซึ้งเธอหรี่ตามองเธมม์…ช่างไร้เดียงสาเสียจริง พี่ชายคนใหม่ของเธอคืนนั้นเธอฝันเห็นฝาแฝดของเธอ “ฉันคิดถึงเธอมากเลย” เธอพูดกับเงาเลือนลางของคนที่มีใบหน้าคล้ายคลึงกับเธอ “ฉันก็คิดถึงเธอเหมือนกัน” เสียงเงาของเธอตอบกลับออกมา หลังจากตื่นเช้ามาวันนั้น น้ำตาเธอท่วมหมอนอีกแล้ว เธอตัดสินใจอ้อนท่านแม่ขอกลับแคว้น อยู่ในเมืองหลวง เธอไม่มีความสุขเอาเสียเลย ขณะนั่งรถม้ากลับ โดยมีเธอ ท่านแม่และเธมม์  “อันนี้ซองอะไรจ๊ะเรนิส” ท่านแม่เธอทักออกมา “ไม่แน่ใจค่ะ เหมือนเป็นซองที่แอนนาวางไว้ให้เมื่อเช้า ยังไม่ทันเปิดดูก็เลยหยิบติดมือมา น่าจะเป็นบัตรเชิญงานเลี้ยงน้ำชาของใครสักคนมั้ง” เรนิสค่อยๆแกะซอง บัตรเชิญเธอและเธมม์ไปงานเลี้ยงน้ำชาของเจ้าชายซีริลตอนบ่ายนี้ ไม่ทันเสียแล้ว เธอถอนหายใจ ชะโงกหน้าบอกคนขับรถม้าให้ส่งใครสักคนรีบไปแจ้งทางวังหลวงว่าพวกเธอเดินทางกลับแคว้นแล้ว ขณะกำลังคิดว่าจะฝากบอกว่าเธอชอบของขวัญของเขามาก แต่นึกขึ้นได้เธอลืมแกะกล่องของขวัญจากเจ้าชาย รู้สึกจะกองๆกันไว้ 7-8 กล่อง เธอมัวแต่คิดเรื่องการออกแบบเวทอวยพรรูปแบบต่างๆจนลืม…ช่างเถิด ปกติเวลาไม่ได้อยู่ในเมืองหลวง เธอแทบจะลืมอยู่แล้วว่ามีเขาอยู่…เธอไม่อยากจะคิดเรื่องเขาให้เครียด ขอเธอมีชีวิตที่มีความสุขไปอีกหลายๆปีน่าจะดีกว่าอย่างน้อย ก่อนที่จะเข้าโรงเรียนเวทมนตร์อะไรนั่นก็ยังดี…
